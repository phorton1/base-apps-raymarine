```perl
sub _classifyE80Items
{
    my ($items) = @_;
    return undef unless @$items;
    my $dbh = connectDB();
    return undef unless $dbh;
    my ($n_present, $n_absent) = (0, 0);
    for my $item (@$items)
    {
        my $uuid = $item->{uuid} // '';
        next unless $uuid;
        my $t = $item->{type} // '';
        my $found;
        if    ($t eq 'waypoint')   { $found = getWaypoint($dbh, $uuid); }
        elsif ($t eq 'group')      { $found = getCollection($dbh, $uuid); }
        elsif ($t eq 'route')      { $found = getRoute($dbh, $uuid); }
        elsif ($t eq 'track')      { $found = getTrack($dbh, $uuid); }
        $found ? $n_present++ : $n_absent++;
    }
    disconnectDB($dbh);
    return 'paste' if $n_present == 0;
    return 'sync'  if $n_absent  == 0;
    return 'mixed';
}
```

```perl
sub _doSync
{
    my ($panel, $right_click_node, $tree, @nodes) = @_;

    my $cb = getClipboard();
    if (!$cb || ($cb->{clipboard_class} // '') ne 'sync')
    {
        warning(0, 0, "IMPLEMENTATION ERROR: _doSync called with non-sync-classified clipboard");
        return;
    }

    my @items = @{$cb->{items} // []};
    if (!@items)
    {
        warning(0, 0, "_doSync: empty clipboard");
        return;
    }

    # Newer-wins pre-flight: check db_version vs e80_version for each item
    # (version wiring partially deferred; for now treat all as plain conflict)
    # If any item's e80_version > db_version, warn and offer proceed/abort.
    # For now, simple confirmation: "Sync N item(s) to DB?"
    my $n   = scalar @items;
    my $msg = "Sync $n item(s) from E80 to database?";
    my $proceed = $nmDialogs::suppress_confirm
        ? 1
        : confirmDialog($tree, $msg, 'Sync');
    return unless $proceed;

    if ($panel eq 'database')
    {
        nmOpsDB::_syncFromE80($right_click_node, $tree, \@items);
    }
    else
    {
        nmOpsE80::_syncToE80($right_click_node, $tree, \@items);
    }
```

### STEP 4 -- nmOpsDB.pm: _syncFromE80

New exported sub. For each item in the sync-classified clipboard, update the DB record
from the E80 snapshot, increment db_version, set e80_version = db_version.

```perl
sub _syncFromE80
{
    my ($right_click_node, $tree, $items) = @_;

    my $dbh = connectDB();
    return unless $dbh;

    for my $item (@$items)
    {
        my $t    = $item->{type} // '';
        my $uuid = $item->{uuid} // '';
        next unless $uuid;

        if ($t eq 'waypoint')
        {
            my $wp = $item->{data} // {};
            updateWaypoint($dbh, $uuid,
                $wp->{name}    // '',
                $wp->{lat}     // 0,
                $wp->{lon}     // 0,
                $wp->{sym}     // 0,
                $wp->{comment} // '');
            my $rec = getWaypoint($dbh, $uuid);
            if ($rec)
            {
                my $new_ver = ($rec->{db_version} // 0) + 1;
                $dbh->do("UPDATE waypoints SET db_version=?, e80_version=? WHERE uuid=?",
                    undef, $new_ver, $new_ver, $uuid);
            }
        }
        elsif ($t eq 'route')
        {
            my $rd  = $item->{data} // {};
            my $rps = $item->{route_points} // [];
            updateRoute($dbh, $uuid,
                $rd->{name}    // '',
                e80RouteIndexToAbgr($rd->{color}),
                $rd->{comment} // '');
            clearRouteWaypoints($dbh, $uuid);
            my $pos = 0;
            appendRouteWaypoint($dbh, $uuid, $_->{uuid}, $pos++) for @$rps;
            my $rec = getRoute($dbh, $uuid);
            if ($rec)
            {
                my $new_ver = ($rec->{db_version} // 0) + 1;
                $dbh->do("UPDATE routes SET db_version=?, e80_version=? WHERE uuid=?",
                    undef, $new_ver, $new_ver, $uuid);
            }
        }
        elsif ($t eq 'group')
        {
            my $gd = $item->{data} // {};
            # Update group shell name if changed
            my $existing = getCollection($dbh, $uuid);
            if ($existing && ($existing->{name}//'') ne ($gd->{name}//''))
            {
                $dbh->do("UPDATE collections SET name=? WHERE uuid=?",
                    undef, $gd->{name}//'', $uuid);
            }
            # Sync each member waypoint
            for my $member (@{$item->{members} // []})
            {
                my $mu = $member->{uuid} // '';
                my $md = $member->{data} // {};
                next unless $mu;
                my $mrec = getWaypoint($dbh, $mu);
                next unless $mrec;
                updateWaypoint($dbh, $mu,
                    $md->{name}//'', $md->{lat}//0, $md->{lon}//0,
                    $md->{sym}//0, $md->{comment}//'');
                my $new_ver = ($mrec->{db_version}//0) + 1;
                $dbh->do("UPDATE waypoints SET db_version=?, e80_version=? WHERE uuid=?",
                    undef, $new_ver, $new_ver, $mu);
            }
        }
        # tracks: read-only on E80, sync-down not applicable
    }

    disconnectDB($dbh);
    _refreshDatabase();
}
```

```perl
sub _destIsDescendantOfClipboard
{
    my ($node, $panel, $clip_uuids) = @_;
    return 0 unless $node;

    # Check the node itself
    return 1 if $clip_uuids->{$node->{uuid} // ''};

    if ($panel eq 'database')
    {
        # Walk parent collection chain
        my $dbh = connectDB();
        return 0 unless $dbh;
        my $parent_uuid = ($node->{data} // {})->{collection_uuid}
                       // ($node->{data} // {})->{parent_uuid};
        while ($parent_uuid)
        {
            if ($clip_uuids->{$parent_uuid})
            {
                disconnectDB($dbh);
                return 1;
            }
            my $coll = getCollection($dbh, $parent_uuid);
            last unless $coll;
            $parent_uuid = $coll->{parent_uuid};
        }
        disconnectDB($dbh);
    }
    else  # e80 -- shallow hierarchy
    {
        my $group_uuid = $node->{group_uuid} // '';
        return 1 if $group_uuid && $clip_uuids->{$group_uuid};
    }
    return 0;
}
```
